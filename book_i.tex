		\begin{prop}
		
		On a given finite straight line (\hbox{\tikz[baseline=-0.5ex]\draw[ultra thick] (0,0) -- (1,0);}) to describe an equlateral triangle.
		\begin{tikzpicture}[scale=1.5,opacity=0.8,remember picture]
		\coordinate (A) at (0,0);
		\coordinate (B) at (1,0);
		\draw[name path=C1,blue,ultra thick] (A) circle (1cm);
		\draw[name path=C2,red,ultra thick] (B) circle (1cm);
		\path [name intersections={of=C1 and C2}];
		\coordinate (C) at (intersection-1);
		
		\draw[ultra thick] (A) -- (B); 
		\draw[ultra thick,red] (B) -- (C);
		\draw[ultra thick,yellow] (C) -- (A);
		\end{tikzpicture}
	\end{prop}
	
	
	\begin{proof}
		Describe \tikz{} and \tikz{} (postulate 3.); draw \tikz{} and \tikz{} (post. 1. ). then will \tikz{} be equilateral. 
		
	\end{proof}
